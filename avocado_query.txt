	date date,
	average_price numeric(3,2),
	total_volume numeric(12,4),
	type char(12),
	year int,
	geography varchar(30)
);

create table calisales (
	date int primary key,
	average_price_organic numeric(10,2),

	total_volume_conventional numeric(20,4),
	value decimal
);
select * from avocados;

select date_part(yyyy,Date) + date_part(mm,date) as newdate
